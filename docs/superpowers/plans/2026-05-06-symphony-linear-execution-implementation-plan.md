# Symphony Linear Execution Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Update Symphony so Linear-backed execution uses the shared tracker boundary, dependency-aware dispatch, and backend-neutral worker helper operations with parity to GitHub Projects v2 for normal worker state flow.

**Architecture:** Keep `WORKFLOW.md` as Symphony's configuration source and keep dispatch selection inside backend-specific `TrackerAdapter` implementations. Extend the existing Linear client and adapter to normalize project milestone, parent/sub-issue, blocker, assignee, comment, and document data into the existing `Issue` and helper response shapes.

**Tech Stack:** Rust, Tokio, reqwest, serde, mockito, existing Symphony `TrackerAdapter`, existing `sym-state` helper script, existing Kata CLI skills generated by the prerequisite CLI Linear Core project.

const QUERY_CANDIDATES: &str = r#"
query SymphonyLinearPoll($projectSlug: String!, $stateNames: [String!]!, $first: Int!, $relationFirst: Int!, $after: String) {
  issues(filter: {project: {slugId: {eq: $projectSlug}}, state: {name: {in: $stateNames}}}, first: $first, after: $after) {
    nodes {
      id
      identifier
      title
      description
      priority
      state { name }
      branchName
      url
      assignee { id }
      labels { nodes { name } }
      inverseRelations(first: $relationFirst) {
        nodes {
          type
          issue { id identifier state { name } }
        }
      }
      children { nodes { id identifier } }
      parent { identifier }
      createdAt
      updatedAt
    }
    pageInfo { hasNextPage endCursor }
  }
}
"#;

const QUERY_CANDIDATES_BY_MILESTONE: &str = r#"
query SymphonyLinearPollByMilestone($projectSlug: String!, $projectMilestoneId: ID!, $stateNames: [String!]!, $first: Int!, $relationFirst: Int!, $after: String) {
  issues(filter: {project: {slugId: {eq: $projectSlug}}, projectMilestone: {id: {eq: $projectMilestoneId}}, state: {name: {in: $stateNames}}}, first: $first, after: $after) {
    nodes {
      id
      identifier
      title
      description
      priority
      state { name }
      branchName
      url
      assignee { id }
      labels { nodes { name } }
      inverseRelations(first: $relationFirst) {
        nodes {
          type
          issue { id identifier state { name } }
        }
      }
      children { nodes { id identifier } }
      parent { identifier }
      createdAt
      updatedAt
    }
    pageInfo { hasNextPage endCursor }
  }
}
"#;

fn candidates_query(config: &TrackerConfig) -> &'static str {
    if config.project_milestone_id.is_some() {

}

            "first": ISSUE_PAGE_SIZE,

        if let Some(cursor) = after_cursor.as_ref() {
            variables["after"] = Value::String(cursor.clone());
        }

        let body = self.graphql(QUERY_CHILD_ISSUES, variables).await?;
        let children = body
            .get("data")
            .and_then(|data| data.get("issue"))
            .and_then(|issue| issue.get("children"))

        let nodes = children
            .get("nodes")

            PageCursorResult::Error => return Err(SymphonyError::LinearMissingEndCursor),

    let issue = mutation
        .get("issue")
        .ok_or(SymphonyError::LinearUnknownPayload)?;
    Ok(LinearFollowupIssue {
        id: issue
            .get("id")
            .and_then(|value| value.as_str())
            .unwrap_or("")
            .to_string(),
        identifier: issue
            .get("identifier")
            .and_then(|value| value.as_str())
            .unwrap_or("")
            .to_string(),
        title: issue
            .get("title")
            .and_then(|value| value.as_str())
            .unwrap_or("")
            .to_string(),
        url: issue
            .get("url")
            .and_then(|value| value.as_str())
            .map(String::from),
    })
}

fn decode_project_context(body: &Value, config: &TrackerConfig) -> Result<LinearIssueContext> {
    let projects = body
        .get("data")
        .and_then(|data| data.get("projects"))
        .and_then(|projects| projects.get("nodes"))
        .and_then(|nodes| nodes.as_array())
        .ok_or(SymphonyError::LinearUnknownPayload)?;
    let [project] = projects.as_slice() else {
        return Err(SymphonyError::Other(format!(
            "tracker.project_slug resolved to {} Linear projects",
            projects.len()
        )));
    };

    let project_id = project
        .get("id")
        .and_then(|value| value.as_str())
        .map(String::from)
        .ok_or(SymphonyError::LinearUnknownPayload)?;
    let team_id = select_project_team_id(project, config.team_key.as_deref())?;
    let project_milestone_id = match config.project_milestone_id.as_deref() {
        Some(id) => Some(id.to_string()),
        None => select_project_milestone_id(project, config.project_milestone_name.as_deref())?,
    };

    Ok(LinearIssueContext {
        team_id,
        project_id: Some(project_id),
        project_milestone_id,
    })
}

fn select_project_team_id(project: &Value, configured_team_key: Option<&str>) -> Result<String> {
    let teams = project
        .get("teams")
        .and_then(|teams| teams.get("nodes"))
        .and_then(|nodes| nodes.as_array())
        .ok_or(SymphonyError::LinearUnknownPayload)?;
    if let Some(team_key) = configured_team_key {
        return teams
            .iter()
            .find(|team| {
                team.get("key")
                    .and_then(|value| value.as_str())
                    .is_some_and(|candidate| candidate == team_key)
            })
            .and_then(|team| team.get("id"))
            .and_then(|id| id.as_str())
            .map(String::from)
            .ok_or_else(|| SymphonyError::Other(format!("Linear team '{team_key}' not found on configured project")));
    }
    let [team] = teams.as_slice() else {
        return Err(SymphonyError::Other(format!(
            "configured Linear project has {} teams; set tracker.team_key",
            teams.len()
        )));
    };
    team.get("id")
        .and_then(|id| id.as_str())
        .map(String::from)
        .ok_or(SymphonyError::LinearUnknownPayload)
}

fn select_project_milestone_id(project: &Value, milestone_name: Option<&str>) -> Result<Option<String>> {
    let Some(milestone_name) = milestone_name else {
        return Ok(None);
    };
    let milestones = project
        .get("projectMilestones")
        .and_then(|milestones| milestones.get("nodes"))
        .and_then(|nodes| nodes.as_array())
        .ok_or(SymphonyError::LinearUnknownPayload)?;
    milestones
        .iter()
        .find(|milestone| {
            milestone
                .get("name")
                .and_then(|value| value.as_str())
                .is_some_and(|candidate| candidate == milestone_name)
        })
        .and_then(|milestone| milestone.get("id"))
        .and_then(|id| id.as_str())
        .map(String::from)
        .map(Some)
        .ok_or_else(|| SymphonyError::Other(format!("Linear project milestone '{milestone_name}' not found")))
}

pub fn helper_required_str(input: &serde_json::Value, field: &str) -> Result<String, String> {
    input
        .get(field)
        .and_then(|value| value.as_str())
        .map(str::trim)
        .filter(|value| !value.is_empty())
        .map(ToString::to_string)
        .ok_or_else(|| format!("helper input field `{field}` must be a non-empty string"))
}

pub fn helper_optional_str(input: &serde_json::Value, field: &str) -> Option<String> {
    input
        .get(field)
        .and_then(|value| value.as_str())
        .map(str::trim)
        .filter(|value| !value.is_empty())
        .map(ToString::to_string)
}

pub fn helper_bool(input: &serde_json::Value, field: &str, default: bool) -> bool {
    input
        .get(field)
        .and_then(|value| value.as_bool())
        .unwrap_or(default)
}

pub async fn run_linear_helper(
    tracker: &TrackerConfig,
    operation: &str,
    input: serde_json::Value,
) -> Result<serde_json::Value, String> {
    let adapter = LinearAdapter::new(LinearClient::new(tracker.clone()));

    match operation {
        "issue.get" => {
            let issue_id = helper_required_str(&input, "issueId")?;
            let issues = adapter
                .fetch_issue_states_by_ids(std::slice::from_ref(&issue_id))
                .await
                .map_err(|err| err.to_string())?;
            let issue = issues
                .into_iter()
                .next()
                .ok_or_else(|| format!("issue not found: {issue_id}"))?;
            let children = if helper_bool(&input, "includeChildren", true) {
                adapter
                    .list_child_issues(&issue_id)
                    .await
                    .map_err(|err| err.to_string())?
            } else {
                Vec::new()
            };
            let comments = if helper_bool(&input, "includeComments", true) {
                adapter
                    .read_document_comments(&issue_id)
                    .await
                    .map_err(|err| err.to_string())?
            } else {
                Vec::new()
            };
            Ok(serde_json::json!({
                "issue": issue,
                "children": children,
                "comments": comments
            }))
        }
        "issue.list-children" => {
            let issue_id = helper_required_str(&input, "issueId")?;
            let children = adapter
                .list_child_issues(&issue_id)
                .await
                .map_err(|err| err.to_string())?;
            Ok(serde_json::json!({ "children": children }))
        }
        "comment.upsert" => {
            let issue_id = helper_required_str(&input, "issueId")?;
            let body = helper_required_str(&input, "body")?;
            let marker = helper_optional_str(&input, "marker");
            let comment = adapter
                .upsert_comment(&issue_id, marker.as_deref(), &body)
                .await
                .map_err(|err| err.to_string())?;
            Ok(serde_json::json!({ "comment": comment }))
        }
        "issue.update-state" => {
            let issue_id = helper_required_str(&input, "issueId")?;
            let state = helper_required_str(&input, "state")?;
            adapter
                .update_issue_state(&issue_id, &state)
                .await
                .map_err(|err| err.to_string())?;
            Ok(serde_json::json!({ "issueId": issue_id, "state": state }))
        }
        "issue.create-followup" => {
            let title = helper_required_str(&input, "title")?;
            let description = helper_required_str(&input, "description")?;
            let parent_issue_id = helper_optional_str(&input, "parentIssueId");
            let issue = adapter
                .create_followup_issue(parent_issue_id.as_deref(), &title, &description)
                .await
                .map_err(|err| err.to_string())?;
            Ok(serde_json::json!({ "issue": issue }))
        }
        "document.read" => {
            let issue_id = helper_required_str(&input, "issueId")?;
            let documents = adapter
                .read_document_comments(&issue_id)
                .await
                .map_err(|err| err.to_string())?;
            if let Some(title) = helper_optional_str(&input, "title") {
                let content = documents
                    .iter()
                    .find(|document| document.title == title)
                    .map(|document| document.content.clone());
                Ok(serde_json::json!({ "title": title, "content": content }))
            } else {
                Ok(serde_json::json!({ "documents": documents }))
            }
        }
        "document.write" => {
            let issue_id = helper_required_str(&input, "issueId")?;
            let title = helper_required_str(&input, "title")?;
            let content = helper_required_str(&input, "content")?;
            let document = adapter
                .write_document_comment(&issue_id, &title, &content)
                .await
                .map_err(|err| err.to_string())?;
            Ok(serde_json::json!({ "document": document }))
        }
        "pr.inspect-checks" | "pr.inspect-feedback" | "pr.land-status" => Err(format!(
            "operation `{operation}` is only available when tracker.kind is github"
        )),
        other => Err(format!(
            "unsupported Symphony helper operation for Linear tracker: {other}"
        )),
    }
}
